trade_union_low_influence = {
	custom_trigger_tooltip = {
		tooltip = trade_union_low_inf
		OR = {
			check_variable = {
				var = trade_union_inf
				compare = less_than_or_equals
				value = 10
			}
			check_variable = {
				var = trade_union_inf
				compare = greater_than_or_equals
				value = 10
			}
		}
	}
}
