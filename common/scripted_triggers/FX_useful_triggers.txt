# check if these three government types for the country in scope
has_socialist_government = {
    custom_trigger_tooltip = {
        tooltip = FX_is_socialist_tt
        OR = {
            has_government = vanguardist
            has_government = orthodox_marxist
            has_government = revolutionary_socialist
            has_government = democratic_socialist
        }
    }
}

has_democratic_government = {
    custom_trigger_tooltip = {
        tooltip = FX_is_democratic_tt
        OR = {
            has_government = social_democrat
            has_government = social_liberal
            has_government = market_liberal
            has_government = social_conservative
            has_government = authoritarian_democrat
        }
    }
}

has_authoritarian_govenrment = {
    custom_trigger_tooltip = {
        tooltip = FX_is_authoritarian_tt
        OR = {
            has_government = authoritarian_democrat
            has_government = paternal_autocrat
            has_government = national_populist
            has_government = valkist
        }
    }
}

can_replace_hog = {
    custom_trigger_tooltip = {
        tooltip = FX_can_replace_hog
        always = no
    }
}

can_replace_minister = {
    custom_trigger_tooltip = {
