has_RightKMT_leader = {
	OR = {
		has_country_leader = {
			name = "Chiang Kai-Shek"
			ruling_only = yes 
		}
		has_country_leader = {
			name = "Li Zongren"
			ruling_only = yes 
		}
		has_country_leader = {
			name = "Long Yun"
			ruling_only = yes 
		}
	}
	NOT = {
		OR = {
			has_government = social_democrat 
			has_government = libertarian_socialist 
			has_government = collectivist
			has_government = vanguardist
		}
	}
}
has_LeftKMT_leader = {
	OR = {
		has_government = social_democrat 
		has_government = libertarian_socialist 
		has_government = collectivist
		has_government = vanguardist
	}
	NOT = {
		OR = {
			has_country_leader = {
				name = "Chiang Kai-Shek"
				ruling_only = yes 
			}
			has_country_leader = {
				name = "Li Zongren"
				ruling_only = yes 
			}
		}
	}
}
