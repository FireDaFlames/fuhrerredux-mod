defined_text = {
	name = GetResLevRank 
	text = {
		trigger = {
			ROOT = {
				check_variable = {
					var = reslev
					value = 45
					compare = greater_than
				}
			}
		}
		localization_key = SIA_resistance_level_dangerous
	}
	text = {
		trigger = {
			ROOT = {
				AND = {
					check_variable = {
						var = reslev
						value = 45
						compare = less_than_or_equals
					}
					check_variable = {
						var = reslev 
						value = 20
						compare = greater_than
					}
				}
			}
		}
		localization_key = SIA_resistance_level_high
	}
	text = {
		trigger = {
			ROOT = {
				AND = {
					check_variable = {
						var = reslev
						value = 20
						compare = less_than_or_equals
					}
					check_variable = {
						var = reslev 
						value = 10
						compare = greater_than
					}
				}
			}
		}
		localization_key = SIA_resistance_level_moderate
	}
	text = {
		trigger = {
			ROOT = {
				AND = {
					check_variable = {
						var = reslev
						value = 10
						compare = less_than_or_equals
					}
					check_variable = {
						var = reslev 
						value = 0
						compare = greater_than_or_equals
					}
				}
			}
		}
		localization_key = SIA_resistance_level_secure
	}
}
