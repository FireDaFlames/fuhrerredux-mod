on_actions = {
	on_startup = {
		effect = {
			random_country = {
				limit = { tag = SIA }
				country_event = {
					id = siam_start.1
					days = 2
				}
			}
		}
	}
	on_new_term_election = {
		effect = {
			if = {
				limit = { tag = SIA }
				country_event = siamelection.1
			}
		}
	}
	on_daily = {
		effect = {
			if = {
				limit = {
					tag = SIA
					OR = {
						has_variable = market_inf
						has_variable = natpop_inf
						has_variable = valkist_inf
					}
				}
				SIA_recalc_party_inf = yes
				SIA_recalc_pop_point = yes
			}
		}
	}
	on_daily = {
		effect = {
			if = {
				limit = {
					tag = SIA
					has_variable = anarzone
				}
			}
			recalc_all_zone = yes 
		}
	}
	on_annex = {
		effect = {
			if = {
				limit = { 
					tag = SIA
					FROM = { tag = UWP } 
				}
				SIA = { 
					country_event = siamcivilwar.22 
					set_country_flag = SIA_win_civil_war
				}
			}
		}
	}
}
