# Philippines on-actions
# created by: dax0102

on_actions = {
  on_new_term_election = {
    effect = {
      if = {
        limit = { tag = PHI }
        country_event = philippines.34 # 1937 Presidential Elections
        country_event = philippines.57 # 1941 Presidential Elections
      }
    }
  }
  on_declare_war = {
    effect = {
      if = {
        limit = {
          ROOT = { tag = PHI }
          FROM = {
            OR = {
              tag = JAP
              tag = ENG
              tag = FRA
              tag = USA
              tag = ITA
              tag = GER
            }
            has_country_flag = PHI_investments
          }
        }

        # Investments are pulled
        ROOT = {
          country_event = { id = philippines.7 days = 2 }
        }
      }
    }
  }
  # ROOT is new controller
  # FROM is old controller
  # FROM.FROM is state id
  on_state_control_changed = {
    effect = {
      if = {
        limit = {
          FROM.FROM = { state = 819 } # Singapore
          FROM = { tag = ENG }
          OR = {
            tag = PHI
            any_other_country = { 
              is_in_faction_with = PHI 
            }
          }
        }

        FROM = { country_event = philippines.76 }
      }
    }
  }
  # ROOT is the capitualated country
  # FROM is winner
  on_capitulation = {
    effect = {
      if = {
        limit = {
          ROOT = { tag = FIC }
          FROM = {
            tag = PHI
            any_other_country = { 
              is_in_faction_with = PHI 
            }
          }
        }

        FRA = { country_event = philippines.77 }
      }
    }
  }
}
