country_event = { # Ikhwan Raid
  id = arabia.1
  title = arabia.1.t
  desc = arabia.1.d
  picture = GFX_report_event_ARB_troops

  trigger = {
    tag = ARB
    country_exists = ARB
    has_idea = ARB_ikhwan_raids
    NOT = { has_war_with = SAU }
  }

  mean_time_to_happen = {
    days = 120
    modifier = {
      factor = 0.75
      check_variable = {
        var = british_control
        value = 50
        compare = greater_than_or_equals
      }
    }
    modifier = {
      factor = 2
      NOT = { controls_state = 1032 }
    }
    modifier = {
      factor = 2
      NOT = { controls_state = 292 }
    }
  }

  option = { # After them!
    name = arabia.1.a
    add_political_power = -5
    add_manpower = -1
  }

  option = { # We'll need make it
    name = arabia.1.b
    add_political_power = -10
  }
}
